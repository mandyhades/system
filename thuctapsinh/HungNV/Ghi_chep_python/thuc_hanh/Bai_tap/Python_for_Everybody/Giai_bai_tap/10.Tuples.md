import modulme
file = modulme.checkfile()
if file != 0:
    counts = dict()
    for line in file:
        words = line.split()
        if line.startswith('From'):
            if line.startswith('From:'):
                continue
            else:
                mail = words[1]
                if mail not in counts: 
                    counts[mail] = 1
                else: 
                    counts[mail] += 1
        else: 
            continue
    print(counts)
else: 
    exit()

tupcounts = list()
for key, val in counts.items():
    tupcounts.append((val, key))
tupcounts.sort(reverse = True)
val, key = max(tupcounts)
print(key, ':', val)
