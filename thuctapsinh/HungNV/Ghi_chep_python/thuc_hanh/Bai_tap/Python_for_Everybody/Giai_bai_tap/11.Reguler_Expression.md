```
import re
import math
name = input("Nhap vao file : ")
try:
    fhand = open(name)
except:
    print("khong phai file")
    exit()

numbers = list()
for line in fhand:
    line = line.rstrip()
    regex = re.findall('^New .* ([0-9]+)', line)
    if len(regex)>0:
        for number in regex:
            number = float(number)
            numbers.append(number)
Tong = sum(numbers)
Count = len(numbers)
print(math.trunc(Tong/Count))
```
